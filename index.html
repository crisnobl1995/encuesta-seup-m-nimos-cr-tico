if (action === 'csv') { return exportCSV(); } // action === 'get' → devuelve JSON con todas las respuestas agrupadas por hospital const ss = getSpreadsheet(); const raw = ss.getSheetByName(SHEET_RAW); const data = raw.getDataRange().getValues(); if (data.length <= 1) { return jsonResponse([]); } // Agrupar por hospital const hospitals = {}; for (let i = 1; i < data.length; i++) { const [ts, hospital, ccaa, tipo, volumen, ucip, item, resp] = data[i]; if (!hospital) continue; const key = hospital + '|' + ts; if (!hospitals[key]) { hospitals[key] = { hospital, ccaa, tipo, volumen, ucip, timestamp: ts, responses: {} }; } hospitals[key].responses[item] = resp; } return jsonResponse(Object.values(hospitals)); } // ── POST: recibe una respuesta de la encuesta ─────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = getSpreadsheet(); const raw = ss.getSheetByName(SHEET_RAW); const ts = data.timestamp || new Date().toISOString(); const hospital = data.hospital || ''; const ccaa = data.ccaa || ''; const tipo = data.tipo || ''; const volumen = data.volumen || ''; const ucip = data.ucip || ''; const rows = Object.entries(data.responses || {}).map(([item, resp]) => [ts, hospital, ccaa, tipo, volumen, ucip, item, resp.toUpperCase()] ); if (rows.length > 0) { raw.getRange(raw.getLastRow() + 1, 1, rows.length, 8).setValues(rows); } updateResumen(ss); return jsonResponse({ ok: true, rows: rows.length }); } catch(err) { return jsonResponse({ ok: false, error: err.message }); } } // ── RESUMEN: actualiza la hoja de resumen ─────────────────────── function updateResumen(ss) { const raw = ss.getSheetByName(SHEET_RAW); const res = ss.getSheetByName(SHEET_RESUMEN); const data = raw.getDataRange().getValues(); if (data.length <= 1) return; const counts = {}; for (let i = 1; i < data.length; i++) { const item = data[i][6]; const resp = (data[i][7] || '').toUpperCase(); if (!item) continue; if (!counts[item]) counts[item] = { SI:0, NO:0, NP:0 }; if (resp === 'SI' || resp === 'SÍ') counts[item].SI++; else if (resp === 'NO') counts[item].NO++; else if (resp === 'NP' || resp === 'N/P') counts[item].NP++; } const rows = Object.entries(counts).map(([item, c]) => { const total = c.SI + c.NO + c.NP; return [item, c.SI, c.NO, c.NP, total, total > 0 ? Math.round(c.SI/total*100)+'%' : '—', total > 0 ? Math.round(c.NO/total*100)+'%' : '—']; }).sort((a,b) => b[2] - a[2]); // ordenar por NO desc res.clearContents(); res.appendRow(['Item','Total_SI','Total_NO','Total_NP','Total_Respuestas','Pct_SI','Pct_NO']); res.getRange(1,1,1,7).setBackground('#0d2340').setFontColor('#ffffff').setFontWeight('bold'); if (rows.length > 0) { res.getRange(2, 1, rows.length, 7).setValues(rows); // Formato condicional: rojo en col NO si alto const noRange = res.getRange(2, 3, rows.length, 1); const rule = SpreadsheetApp.newConditionalFormatRule() .whenNumberGreaterThan(0) .setBackground('#fce8e6') .setFontColor('#c0392b') .setRanges([noRange]) .build(); res.setConditionalFormatRules([rule]); } } // ── CSV EXPORT ────────────────────────────────────────────────── function exportCSV() { const ss = getSpreadsheet(); const raw = ss.getSheetByName(SHEET_RAW); const data = raw.getDataRange().getValues(); const lines = data.map(row => row.map(cell => `"${String(cell).replace(/"/g,'""')}"`).join(';') ); return ContentService .createTextOutput(lines.join('\n')) .setMimeType(ContentService.MimeType.TEXT); } // ── HELPERS ───────────────────────────────────────────────────── function getSpreadsheet() { const files = DriveApp.getFilesByName(SPREADSHEET_NAME); if (files.hasNext()) return SpreadsheetApp.open(files.next()); // Si no existe, créala (primera vez después del setup) const ss = SpreadsheetApp.create(SPREADSHEET_NAME); setup(); return ss; } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
